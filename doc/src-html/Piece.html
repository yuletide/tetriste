095                    public String toString(int o){
096                            int[][] temp = templates[o];
097                            String ret = "";
098                            for (int i=0; i<temp.length; i++){
099                                    for (int j=0; j<temp[0].length; j++){
100                                            ret+=temp[i][j];
101                                    }
102                                    ret+="\n";
103                            }
104                            return ret;
105                    }
