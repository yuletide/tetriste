212            public void initBoard() {
213                    board = new Vector<Vector<Integer>>(unitWidth);
214                    for (int i = 0; i < unitWidth; i++) { // populate board with EMPTY
215                            // values (0)
216                            board.add(i, new Vector<Integer>(unitHeight));
217                            for (int j = 0; j < unitHeight; j++) {
218                                    // System.out.print(" "+(i+width*j)+ " " );
219                                    board.get(i).add(GameBoard.EMPTY);
220                            }
221                    }
222            }

267             * Zero-indexed cell lookup function
268             * 
269             * @param x
270             * @param y
271             * @return
272             */
273            public synchronized Integer getCell(int x, int y) {
274                    if (x > -1 && x < unitWidth && y < unitHeight && board.size() > 0)
275                            return board.get(x).get(y);
276                    else {
277                            System.err
278                            .println("x>-1 && x<unitWidth && y<unitHeight && board.size()>0   "
279                                            + (x > -1)
280                                            + (x)
281                                            + (unitWidth)
282                                            + (y)
283                                            + (unitHeight)
284                                            + (board.size() > 0));
285                            throw new ArrayIndexOutOfBoundsException();
286    
287                    }
288            }

316            /**
317             * Check board for, then clear, completed lines
318             */
319            public void checkBoard() {
320                    System.out.println("GameBoard.checkBoard()");
321                    loseTest();
322                    int clearedLines = 0;
323                    //System.out.println("GameBoard.checkLines() " + unitHeight);
324                    for (int j = 0; j < unitHeight; j++) {
325                            //      System.out.println("Checking row " + j);
326                            if (checkRow(j)) {
327                                    clearedLines++;
328                                    clearRow(j);
329                            }
330                    }
331                    lines += clearedLines;
332                    apl.setLInes(lines);
333                    switch (clearedLines) {
334                    case 0:
335                            break;
336                    case 1:
337                            score(level * 100);
338                            break;
339                    case 2:
340                            score(level * 300);
341                            break;
342                    case 3:
343                            score(level * 500);
344                            break;
345                    case 4:
346                            score(level * 800);
347                            break;
348                    default:
349                            break;
350                    }
351                    System.out.println(score);
352            }

473            /**
474             * Pause or unpause the game
475             * 
476             * @return the resulting value of <code>paused</code>()
477             */
478            public boolean pauseToggle() {
479                    if (isPlaying() && !paused) {
480                            dropTimer.stop();
481                            timer.stop();
482                            paused = !paused;
483                    } else if (isPlaying()) {
484                            dropTimer.start();
485                            timer.start();
486                            paused = !paused;
487                    }
488                    return paused;
489            }

506            /**
507             * Return whether a cell value is equal to the EMPTY value, or False if
508             * query is out of bounds (for move validity checking)
509             * 
510             * @param x
511             * @param y
512             * @return
513             */
514            public boolean isEmpty(int x, int y) {
515                    // System.out.println("GameBoard.isEmpty()"+x+" "+y+" "+(getCell(x,
516                    // y)==EMPTY));
517                    if (y < getUnitHeight() && x < getUnitWidth())
518                            return getCell(x, y) == EMPTY;
519                    else
520                            return false;
521            }

564             * Add a piece to the center of the top of the game board (by filling in the
565             * corresponding board cells with the piece's type value)
566             * 
567             * @param p
568             *            the piece to add
569             */
570            public void add(Piece p) {
571                    System.out.println("GameBoard.add(Piece p) gb:"+p.gb+" "+Piece.Shape.values()[0].getWidth(0));
572                    thePiece = p;
573                    try {
574                            thePiece.setX((unitWidth-p.getLocalWidth())/ 2);
575                            thePiece.setY(0);
576                            hasChanged(true);
577                    } catch (Exception e) {
578                            System.out.println("ERROR adding piece, game over!");
579                            apl.setTitle("GAME OVER");
580                            endGame();
581                            //System.out.println(apl.leninIcon.getImageLoadStatus());
582                            apl.getGraphics().drawImage(apl.lenin, this.getX(), this.getY(), this.getWidth(), this.getHeight(), this);
583                    }
584            }
